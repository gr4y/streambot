=== A simple example how to use streambot

	require 'streambot'

	@auth = {:username=>'username',:password=>'password'}
	@blacklist = ['mac_rt','apple_rt']
	bot = StreamBot::Tracker.new(@auth, @blacklist, 'apple','ipad','iphone os 4','steve jobs')
	bot.start
